Get the JSON-RPC endpoint:
{{- if contains "NodePort" .Values.reth.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "reth.fullname" . }}-reth)
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "RPC URL: http://$NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.reth.service.type }}
  NOTE: It may take a few minutes for the LoadBalancer IP to be available.
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "reth.fullname" . }}-reth --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "RPC URL: http://$SERVICE_IP:{{ include "reth.httpPort" . }}"
{{- else }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "reth.fullname" . }}-reth {{ include "reth.httpPort" . }}:{{ include "reth.httpPort" . }}
  echo "RPC URL: http://127.0.0.1:{{ include "reth.httpPort" . }}"
{{- end }}
